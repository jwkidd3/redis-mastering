Cache-aside Pattern

Most Common Pattern: Application manages both cache and database

How It Works

  1. Read: Check cache first, fallback to database
  2. Write: Update database, then invalidate cache
  3. Miss: Load from database, populate cache

Cache-aside Implementation

JavaScript Implementation


async function getCustomerProfile(customerId) {
    const cacheKey = `customer:${customerId}`;
    
    // Try cache first
    let customer = await redis.get(cacheKey);
    if (customer) {
        return JSON.parse(customer);
    }
    
    // Cache miss - load from database
    customer = await database.getCustomer(customerId);
    
    // Populate cache with TTL
    await redis.setex(cacheKey, 3600, JSON.stringify(customer));
    
    return customer;
}

Write-through Caching

Synchronous Pattern: Write to cache and database simultaneously

Implementation Example


async function updateCustomerProfile(customerId, updates) {
    const cacheKey = `customer:${customerId}`;
    
    // Update database first
    const updatedCustomer = await database.updateCustomer(
        customerId, updates
    );
    
    // Update cache
    await redis.setex(cacheKey, 3600, 
        JSON.stringify(updatedCustomer)
    );
    
    return updatedCustomer;
}

Write-through Benefits & Considerations

When to Use

  • Strong consistency required
  • Read-heavy workloads
  • Critical data updates

Considerations

  • Higher write latency
  • Cache always current
  • Complex error handling

Write-behind (Write-back) Caching

Asynchronous Pattern: Cache updated immediately, database later

Implementation Approach


async function updateCustomerQuick(customerId, updates) {
    const cacheKey = `customer:${customerId}`;
    
    // Update cache immediately
    await redis.setex(cacheKey, 3600, JSON.stringify(updates));
    
    // Queue for database update
    await redis.lpush('db_update_queue', JSON.stringify({
        table: 'customers',
        id: customerId,
        data: updates,
        timestamp: Date.now()
    }));
    
    return { success: true, cached: true };
}

Write-behind Background Processing

Background Worker


async function processDatabaseUpdates() {
    const update = await redis.brpop('db_update_queue', 0);
    const updateData = JSON.parse(update[1]);
    
    try {
        await database.updateCustomer(updateData.id, updateData.data);
        console.log(`Updated customer ${updateData.id}`);
    } catch (error) {
        // Add to dead letter queue for retry
        await redis.lpush('failed_updates', update[1]);
    }
}

// Start background worker
processDatabaseUpdates();

Event-driven Cache Invalidation

New Approach: Leverage Redis Streams for cache coherence

Stream-based Cache Invalidation


// Publish cache invalidation event
async function invalidateCache(cacheKeys, reason) {
    await redis.xAdd('cache:invalidation', '*', {
        keys: JSON.stringify(cacheKeys),
        reason: reason,
        timestamp: Date.now()
    });
}

// Listen for invalidation events
async function setupCacheInvalidationListener() {
    const consumer = redis.duplicate();
    
    while (true) {
        const events = await consumer.xReadGroup(
            'cache-invalidators', 'worker-1',
            [{ key: 'cache:invalidation', id: '>' }],
            { COUNT: 10, BLOCK: 1000 }
        );
        
        for (const event of events[0].messages) {
            const keys = JSON.parse(event.message.keys);
            await redis.del(...keys);
        }
    }
}

Tag-based Cache Invalidation

Implementation Pattern


async function setCacheWithTags(key, value, ttl, tags) {
    // Store the data
    await redis.setex(key, ttl, JSON.stringify(value));
    
    // Associate with tags
    for (const tag of tags) {
        await redis.sadd(`tag:${tag}`, key);
        await redis.expire(`tag:${tag}`, ttl + 300); // Longer TTL
    }
}

async function invalidateByTag(tag) {
    // Get all keys with this tag
    const keys = await redis.smembers(`tag:${tag}`);
    
    if (keys.length > 0) {
        // Delete all tagged keys
        await redis.del(...keys);
        // Clean up tag set
        await redis.del(`tag:${tag}`);
    }
}

Multi-level Caching Architecture

L1: Application Memory

  • Fastest access
  • Limited capacity
  • Process-specific
  • No network overhead

L2: Redis Cache

  • Shared across instances
  • Network overhead
  • Larger capacity
  • Persistent storage

L3: Database

  • Source of truth
  • Highest latency
  • Full dataset
  • ACID guarantees

Multi-level Cache Implementation

Implementation Pattern


class MultiLevelCache {
    constructor() {
        this.l1Cache = new Map(); // In-memory
        this.redis = redis;       // L2 cache
    }
    
    async get(key) {
        // L1 check
        if (this.l1Cache.has(key)) {
            return this.l1Cache.get(key);
        }
        
        // L2 check
        const l2Data = await this.redis.get(key);
        if (l2Data) {
            const parsed = JSON.parse(l2Data);
            this.l1Cache.set(key, parsed); // Populate L1
            return parsed;
        }
        
        // L3 fallback (database)
        return null;
    }
}

Key Strategies

Reduce Network Calls

Error Handling & Resilience

Circuit Breaker Pattern


class CacheCircuitBreaker {
    constructor(threshold = 5, timeout = 60000) {
        this.failures = 0;
        this.threshold = threshold;
        this.timeout = timeout;
        this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
        this.nextAttempt = Date.now();
    }
    
    async execute(operation) {
        if (this.state === 'OPEN') {
            if (Date.now() < this.nextAttempt) {
                throw new Error('Circuit breaker is OPEN');
            }
            this.state = 'HALF_OPEN';
        }
        
        try {
            const result = await operation();
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }
}

Graceful Degradation

Fallback Strategies


async function getCustomerWithFallback(customerId) {
    try {
        // Try cache first
        const cached = await redis.get(`customer:${customerId}`);
        if (cached) return JSON.parse(cached);
    } catch (cacheError) {
        console.warn('Cache unavailable, using database');
    }
    
    // Fallback to database
    const customer = await database.getCustomer(customerId);
    
    // Try to populate cache (non-blocking)
    setImmediate(async () => {
        try {
            await redis.setex(`customer:${customerId}`, 
                3600, JSON.stringify(customer));
        } catch (error) {
            // Log but don't fail
            console.error('Cache population failed:', error);
        }
    });
    
    return customer;
}

Integration with Event Streams

Modern Pattern: Use Redis Streams for cache coherence and event-driven updates

Stream-powered Cache Updates


// Listen to claim events from Lab 8
async function setupClaimCacheUpdates() {
    const streams = await redis.xReadGroup(
        'cache-updaters', 'worker-1',
        [{ key: 'claims:events', id: '>' }],
        { COUNT: 10, BLOCK: 1000 }
    );
    
    for (const stream of streams) {
        for (const message of stream.messages) {
            await updateClaimCache(message);
        }
    }
}

async function updateClaimCache(message) {
    const eventType = message.message.eventType;
    const claimId = message.message.claimId;
    
    if (eventType === 'claim.statusChanged') {
        // Invalidate claim cache
        await redis.del(`claim:${claimId}`);
        await invalidateByTag(`customer:${message.message.customerId}`);
    }
}
